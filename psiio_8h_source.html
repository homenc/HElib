27 // Reads in encrypted database from file
28 inline helib::Database<helib::Ctxt> readDbFromFile(
29  const std::string& databaseFilePath,
30  const sharedContext& contextp,
31  const helib::PubKey& pk)
32 {
33  // Read in ctxt file header
34  std::ifstream databaseFile(databaseFilePath);
35  if (!databaseFile.is_open()) {
36  throw std::runtime_error("Could not open file '" +
37  databaseFilePath + "'.");
38  }
39 
40  helib::Ctxt zero_ctxt(pk);
41  Reader<helib::Ctxt> reader(databaseFilePath, zero_ctxt);
42 
43  std::pair<long, long> dims = {reader.getTOC().getRows(),
44  reader.getTOC().getCols()};
45 
46  helib::Matrix<helib::Ctxt> data(zero_ctxt, dims.first, dims.second);
47 
48  // Read in ctxts
49  if (dims.second == 1) { // Column vector
50  for (long i = 0; i < dims.first * dims.second; ++i)
51  reader.readDatum(data(i, 0), i, 0);
52  } else {
53  for (long i = 0; i < dims.first; ++i) {
54  for (long j = 0; j < dims.second; ++j) {
55  reader.readDatum(data(i, j), i, j);
56  }
57  }
58  }
59 
60  return helib::Database<helib::Ctxt>(data, contextp);
61 }

63 // Reads in encrypted query from file
64 inline helib::Matrix<helib::Ctxt> readQueryFromFile(
65  const std::string& queryFilePath,
66  const helib::PubKey& pk)
67 {
68  // Read in ctxt file header
69  std::ifstream queryFile(queryFilePath);
70  if (!queryFile.is_open()) {
71  throw std::runtime_error("Could not open file '" +
72  queryFilePath + "'.");
73  }
74 
75  helib::Ctxt zero_ctxt(pk);
76  Reader<helib::Ctxt> reader(queryFilePath, zero_ctxt);
77 
78  std::pair<long, long> dims = {reader.getTOC().getRows(),
79  reader.getTOC().getCols()};
80 
81  if (dims.first != 1 && dims.second != 1) {
82  std::cerr << "Query must be either a row or column vector." << std::endl;
83  exit(EXIT_FAILURE);
84  }
85 
86  helib::Matrix<helib::Ctxt> query(zero_ctxt, dims.first, dims.second);
87 
88  // Read in ctxts
89  // FIXME: Always builds query as a row vector
90  if (dims.first == 1) { // Read in row vector
91  for (long i = 0; i < dims.first * dims.second; ++i)
92  reader.readDatum(query(0, i), 0, i);
93  } else if (dims.second == 1) { // Read in column vector
94  for (long i = 0; i < dims.first * dims.second; ++i) {
95  reader.readDatum(query(i, 0), i, 0);
96  }
97  query.transpose();
98  } else {
99  throw std::runtime_error("Trying to read in query that is not a vector. Dimensions " + std::to_string(dims.first) + " by " + std::to_string(dims.second) + ".");
100  }
101 
102  return query;
103 }
