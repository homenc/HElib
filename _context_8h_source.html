118  // NOTE: this is a bit heuristic: we assume that if we evaluate
119  // f at a primitive root of unity, then we get something that well
120  // approximates a normal random variable with the same variance,
121  // which is equal to the sum of the variances of the individual
122  // f_i's, which is (2*magBound)^2/12 = magBound^2/3.
123  // We then multiply the sqrt of the variance by scale to get
124  // the high probability bound.
125 
126  double noiseBoundForUniform(double magBound, long degBound) const
127  {
128  return scale * std::sqrt(double(degBound) / 3.0) * magBound;
129  }

145  // NOTE: for odd modulus, this means each f_i is uniformly distributed
146  // over { -floor(modulus/2), ..., floor(modulus/2) }.
147  // For even modulus, this means each f_i is uniformly distributed
148  // over { modulus/2, ..., modulus/2 }, except that the two endpoints
149  // (which represent the same residue class) occur with half the
150  // probability of the others.
151 
152  // NOTE: this is a bit heuristic: we assume that if we evaluate
153  // f at a primitive root of unity, then we get something that well
154  // approximates a normal random variable with the same variance,
155  // which is equal to the sum of the variances of the individual
156  // f_i's, which is (modulus)^2/12 + 1/6 for even modulus,
157  // and is at most (modulus^2)/12 for odd modulus.
158  // We then multiply the sqrt of the variance by scale to get
159  // the high probability bound.
160 
161  // NOTE: this is slightly more accurate that just calling
162  // noiseBoundForUniform with magBound=modulus/2.
163 
164  double noiseBoundForMod(long modulus, long degBound) const
165  {
166  double var = fsquare(modulus) / 12.0;
167  if (modulus % 2 == 0)
168  var += 1.0 / 6.0;
169 
170  return scale * std::sqrt(degBound * var);
171  }

183  // then we get a normal random variable variance degBound * sigma^2.
184  // We then multiply the sqrt of the variance by scale to get
185  // the high probability bound.

187  double noiseBoundForGaussian(double sigma, long degBound) const
188  {

251  double stdDevForRecryption(long skHwt = 0) const
252  {
253  if (!skHwt)
254  skHwt = rcData.skHwt;
255  // the default reverts to rcData.skHwt, *not* rcData.defSkHwt
256 
257  long k = zMStar.getNFactors();
258  // number of prime factors of m
259 
260  long m = zMStar.getM();
261  long phim = zMStar.getPhiM();
262 
263  double mrat = double(phim) / double(m);
264 
265  return std::sqrt(mrat * double(skHwt) * double(1L << k) / 3.0) * 0.5;
266  }
