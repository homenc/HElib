140  bool mul(const SKHandle& a, const SKHandle& b)
141  {
142  // If either of the inputs is one, the output equals to the other input
143  if (a.isOne()) {
144  *this = b;
145  return (b.secretKeyID >= 0);
146  }
147  if (b.isOne()) {
148  *this = a;
149  return (a.secretKeyID >= 0);
150  }
151 
152  if (a.secretKeyID == -1 || b.secretKeyID == -1) {
153  secretKeyID = -1; // -1 will be used to indicate an "error state"
154  return false;
155  }
156 
157  if (a.secretKeyID != b.secretKeyID) {
158  secretKeyID = -1;
159  return false;
160  }
161 
162  if (a.powerOfX != b.powerOfX) {
163  secretKeyID = -1;
164  return false;
165  }
166 
167  secretKeyID = a.secretKeyID;
168  powerOfX = a.powerOfX;
169  powerOfS = a.powerOfS + b.powerOfS;
170  return true;
171  }
